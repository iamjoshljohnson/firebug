namespace Fire\Bug;

/**
 * This class provides a standard format for Debuggers
 * when a developer calls the debugger() method.
 */
class Debugger
{

    /**
     * The debugger value
     * @var mixed
     */
    private $_value;

    /**
     * The stack trace to the debugger.
     * @var array
     */
    private $_trace;

    /**
     * The constructor
     */
    public function __construct()
    {
        $this->_value = '';
        $this->_trace = '';
    }

    /**
     * Set the message value.
     * @param mixed $value The message value
     */
    public function setValue($value)
    {
        $this->_value = $value;
    }

    /**
     * Gets the message value.
     * @return mixed
     */
    public function getValue()
    {
        return $this->_value;
    }

    /**
     * Sets the trace. Expected value is the array
     * generated by debug_backtrace().
     * @param mixed[] $trace The array generated by debug_backtrace()
     */
    public function setTrace($trace)
    {
        $this->_trace = $trace;
    }

    /**
     * Gets the trace.
     * @return mixed[]
     */
    public function getTrace()
    {
        return $this->_trace;
    }

}
